zend_module yapısı

PHP uzantısının ana kaynak kodu C programcısı için yeni yapılar içerir. Bunlardan en önemlisi yeni uzantıda ilk dokunulan zend_module yapısıdır. Bu yapı Zen Makinesine sağlıklı bilgi aktarır. Uzantının bağlılıklarını, sürümünü, beri beslemesini ve diğer kritik verilerini belirtir. Zaman içinde yapı çok değişime uğramıştır. PHP.5.0 sürümünden beri görünen biçimi burada kullanılmıştır. Çokaz bir değişiklik içeren 5.1 ve 5.2 bölümleri de belirtilecektir.

example.c kodundan alınan zend_module tanımı hiçbir kod eklemeden aşağıdaki gibi görünür (örnek dosya ext_skel --extname=example ile yaratıldı).

Örnek-1 örnek uzantıda zend_module tanımıi

/* {{{ example_module_entry
 */
zend_module_entry example_module_entry = {
#if ZEND_MODULE_API_NO >= 20010901
    STANDARD_MODULE_HEADER,
#endif
    "example",
    example_functions,
    PHP_MINIT(example),
    PHP_MSHUTDOWN(example),
    /* start isteğinde yapacak birşey yoksa NULL yerleştir */
    PHP_RINIT(example),
    /* end isteğinde yapacak birşey yoksa NULL yerleştir */
    PHP_RSHUTDOWN(example),
    PHP_MINFO(example),
#if ZEND_MODULE_API_NO >= 20010901
    "0.1", /* Sizin kendi uzantınızın sürümünü yazın */
#endif
    STANDARD_MODULE_PROPERTIES
};
/* }}} */
İlk anda biraz tuhaf gelebilir ama birçoğunu anlamak çok kolaydır. Burada zend_module.h dosyasından alınan zend_module tanımı var. Bazı en yakın değişmezler de eklenmiştir.

Örnek-2: PHP 5.2 içinde zend_module tanımı

#define ZEND_MODULE_API_NO 20060613
struct _zend_module_entry {
    unsigned short size;
    unsigned int zend_api;
    unsigned char zend_debug;
    unsigned char zts;
    struct _zend_ini_entry *ini_entry;
    struct _zend_module_dep *deps;
    char *name;
    struct _zend_function_entry *functions;
    int (*module_startup_func)(INIT_FUNC_ARGS);
    int (*module_shutdown_func)(SHUTDOWN_FUNC_ARGS);
    int (*request_startup_func)(INIT_FUNC_ARGS);
    int (*request_shutdown_func)(SHUTDOWN_FUNC_ARGS);
    void (*info_func)(ZEND_MODULE_INFO_FUNC_ARGS);
    char *version;
    size_t globals_size;
#ifdef ZTS
    ts_rsrc_id* globals_id_ptr;
#else
    void* globals_ptr;
#endif
    void (*globals_ctor)(void *global TSRMLS_DC);
    void (*globals_dtor)(void *global TSRMLS_DC);
    int (*post_deactivate_func)(void);
    int module_started;
    unsigned char type;
    void *handle;
    int module_number;
};
#define STANDARD_MODULE_HEADER_EX
    sizeof(zend_module_entry), ZEND_MODULE_API_NO, ZEND_DEBUG, USING_ZTS
#define STANDARD_MODULE_HEADER \
    STANDARD_MODULE_HEADER_EX, NULL, NULL
#define ZE2_STANDARD_MODULE_HEADER \
    STANDARD_MODULE_HEADER_EX, ini_entries, NULL

#define STANDARD_MODULE_PROPERTIES_EX 0, 0, NULL, 0
